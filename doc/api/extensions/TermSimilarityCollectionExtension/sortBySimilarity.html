List<TermSimilarity> sortBySimilarity([bool descending = true]) {
  final list = List<TermSimilarity>.from(this);
  if (descending) {
    list.sort(((a, b) => b.compareTo(a)));
  } else {
    list.sort(((a, b) => a.compareTo(b)));
  }
  return list;
}
