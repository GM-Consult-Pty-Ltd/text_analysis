Map<Term, double> jaccardSimilarityMap(Iterable<Term> terms, [int k = 2]) {
  final retVal = <Term, double>{};
  final termGrams = kGrams(k);
  for (final other in terms) {
    retVal[other] = _jaccardSimilarity(termGrams, other, k);
  }
  return retVal;
}
