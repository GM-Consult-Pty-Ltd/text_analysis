@override
double keyWordScore(List<String> keyWord) {
  double retVal = 0.0;
  for (final e in keyWord) {
    final tF = termFrequency(e);
    final tD = termDegree(e);
    final tV = tD / tF;
    retVal += tV;
  }
  return retVal;
}
